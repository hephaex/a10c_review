{
	struct mct_clock_event_device *mevt;

	// smp_processor_id(): 0
	unsigned int cpu = smp_processor_id();
	// cpu: 0

	// evt: [pcp0] &(&percpu_mct_tick)->evt

	if (!cpumask_equal(newdev->cpumask, cpumask))

	// newdev: [pcp0] &(&percpu_mct_tick)->evt, cpu: 0
